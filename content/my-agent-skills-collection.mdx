### ตัวอย่างโค้ดที่น่าสนใจ

#### ApiError — Global Error Type ที่สะอาดที่สุด

```rust
pub enum ApiError {
    InternalServerError(String),
    NotFound(String),
    BadRequest(String),
    Forbidden(String),
}

// สอน Axum วิธีแปลง Error เป็น HTTP Response
impl IntoResponse for ApiError {
    fn into_response(self) -> Response {
        let (status, message) = match self {
            ApiError::InternalServerError(msg) => (StatusCode::INTERNAL_SERVER_ERROR, msg),
            ApiError::NotFound(msg)            => (StatusCode::NOT_FOUND, msg),
            ApiError::BadRequest(msg)          => (StatusCode::BAD_REQUEST, msg),
            ApiError::Forbidden(msg)           => (StatusCode::FORBIDDEN, msg),
        };
        (status, Json(json!({ "error": message }))).into_response()
    }
}

// ✨ From trait: ใช้ ? operator กับ sqlx query ได้โดยตรง
impl From<sqlx::Error> for ApiError {
    fn from(err: sqlx::Error) -> Self {
        eprintln!("🔥 DB Error: {:?}", err);
        ApiError::InternalServerError("เกิดข้อผิดพลาดในการเชื่อมต่อฐานข้อมูล".to_string())
    }
}
```

#### Dynamic QueryBuilder — Pagination + Multi-Filter

```rust
let mut query_builder: QueryBuilder<Postgres> = QueryBuilder::new(
    "SELECT r.id, r.req_code FROM requests r WHERE r.is_deleted = FALSE"
);

// Filter ตาม search text (SQL injection safe)
if let Some(search) = &params.search {
    let pattern = format!("%{}%", search.to_lowercase());
    query_builder.push(" AND LOWER(r.description) LIKE ");
    query_builder.push_bind(pattern); // ✅ parameterized
}

// Filter array ด้วย ANY($N)
if let Some(ids_str) = &params.status_ids {
    let ids: Vec<i32> = ids_str.split(',')
        .filter_map(|s| s.parse().ok())
        .collect();
    if !ids.is_empty() {
        query_builder.push(" AND r.status_id = ANY(");
        query_builder.push_bind(ids); // Vec<i32> → PostgreSQL array
        query_builder.push(")");
    }
}

// Pagination
query_builder.push(" ORDER BY r.priority ASC LIMIT ");
query_builder.push_bind(limit);
query_builder.push(" OFFSET ");
query_builder.push_bind(offset);
```

#### JWT Middleware — Claims ผ่าน Extensions

```rust
// middleware.rs
pub async fn auth(mut req: Request, next: Next) -> Result<Response, StatusCode> {
    let token = req.headers()
        .get(header::AUTHORIZATION)
        .and_then(|h| h.to_str().ok())
        .filter(|h| h.starts_with("Bearer "))
        .map(|h| &h[7..])
        .ok_or(StatusCode::UNAUTHORIZED)?;

    let claims = decode::<Claims>(token, &key, &Validation::default())
        .map_err(|_| StatusCode::UNAUTHORIZED)?.claims;

    // ✨ ฝัง Claims ไว้ใน request extensions
    req.extensions_mut().insert(claims);
    Ok(next.run(req).await)
}

// handler — ดึง Claims ออกมาใช้
pub async fn create_request(
    State(pool): State<PgPool>,
    Extension(claims): Extension<Claims>, // ← ดึงจาก middleware
    Json(payload): Json<CreateRequestPayload>,
) -> Result<..., ApiError> {
    let user_id: i32 = claims.sub.parse()
        .map_err(|_| ApiError::InternalServerError("Token ไม่ถูกต้อง".to_string()))?;
}
```

#### Database Transaction

```rust
let mut tx = pool.begin().await.map_err(|e| {
    ApiError::InternalServerError(format!("ไม่สามารถเริ่ม Transaction: {}", e))
})?;

// INSERT หลาย table — ใช้ &mut *tx ทั้งหมด
let id: (i32,) = sqlx::query_as(INSERT_SQL)
    .bind(&payload.description)
    .fetch_one(&mut *tx).await?; // ← ต้องใช้ tx ไม่ใช่ pool

for approver in &payload.approvers {
    sqlx::query(INSERT_APPROVAL_SQL)
        .bind(id.0)
        .bind(approver.approver_id)
        .execute(&mut *tx).await?;
}

tx.commit().await?; // ✅ ทุก INSERT สำเร็จ → commit
// ถ้า error ก่อน commit → rollback อัตโนมัติ
```
